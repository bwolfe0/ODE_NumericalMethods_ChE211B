theta = np.arange(0,2*np.pi+np.pi/100,np.pi/100)

z = (1*exp(2*(1j*theta)) - 1) / (.5*exp(1j*theta) + 1.5)
z_real = z.real
z_imag = z.imag

plt.fill(z_real,z_imag, label='Absolute Stability Region')
plt.axhline(y=0, color='k')
plt.axvline(x=0, color='k')
plt.xlim(-2,1)
plt.ylim(-1,1)
plt.xlabel('Re(h*lambda)')
plt.ylabel('Im(h*lambda)')
plt.legend()
plt.title('Absolute Stability Region for Given Arbitrary Method')
plt.show()
